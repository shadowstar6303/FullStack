let x=0;

document.addEventListener("keydown",event=>{
	if(event.key.startswith(("Arrow")){
	
	switch(event.key){
		case "ArrowUp":
		     y -=moveAmount;
		     break;
                case "ArrowDown":
		     y +=moveAmount;
		     break; 		
		case "ArrowLeft":
		     x -=moveAmount;
		     break;
		case "Arrow+Up":
		     x +=moveAmount;
		     break;
	}
	
	myBox.style.top=`${y}px`;
	myBox.style.left=`${x}px`;
	
});
