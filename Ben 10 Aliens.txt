  <script>
    // Dark/Light Mode Toggle
    const toggleBtn = document.getElementById('toggle-mode');
    const body = document.getElementById('page-body');

    toggleBtn.addEventListener('click', () => {
      body.classList.toggle('bg-gray-900');
      body.classList.toggle('bg-white');
      body.classList.toggle('text-white');
      body.classList.toggle('text-black');

      // Change buttons text colors on mode toggle
      document.querySelectorAll('.alien-button').forEach(btn => {
        if (body.classList.contains('bg-white')) {
          btn.classList.add('text-black');
          btn.classList.remove('text-white');
          btn.classList.add('hover:brightness-90');
          btn.classList.remove('hover:brightness-110');
        } else {
          // Reset colors based on background
          btn.classList.remove('text-black');
          btn.classList.add('text-white');
          btn.classList.remove('hover:brightness-90');
          btn.classList.add('hover:brightness-110');
        }
      });

      // Update aria-pressed
      let pressed = toggleBtn.getAttribute('aria-pressed') === 'true';
      toggleBtn.setAttribute('aria-pressed', !pressed);
    });

    // Play sound & animation on button click
    const clickSound = document.getElementById('click-sound');

    document.querySelectorAll('.alien-button').forEach(button => {
      button.addEventListener('click', () => {
        // Play sound
        clickSound.currentTime = 0;
        clickSound.play();

        // Add a quick "pressed" animation class to simulate click feedback
        button.classList.add('active');
        setTimeout(() => {
          button.classList.remove('active');
        }, 150);
      });
    });
  </script>

    const menu = document.getElementById('menu');
    const highlight = document.querySelector('.highlight');

    function moveHighlight(el) {
      const rect = el.getBoundingClientRect();
      const parentRect = menu.getBoundingClientRect();

      highlight.style.width = `${rect.width}px`;
      highlight.style.left = `${rect.left - parentRect.left}px`;
    }

    // Set highlight initially on the active menu item
    const activeLink = menu.querySelector('a.active');
    if (activeLink) {
      moveHighlight(activeLink);
    }

    // Move highlight on hover
    menu.querySelectorAll('a').forEach(link => {
      link.addEventListener('mouseenter', () => moveHighlight(link));
      link.addEventListener('mouseleave', () => {
        if (activeLink) moveHighlight(activeLink);
        else highlight.style.width = 0;
      });
    });
